import pandas as pd
from pathlib import Path

def load_excel(file_path):
    file_ext = Path(file_path).suffix.lower()

    if file_ext == ".xlsx":
        return pd.read_excel(file_path, engine="openpyxl", header=1)

    elif file_ext == ".xls":
        return pd.read_excel(file_path, engine="xlrd")

    else:
        raise ValueError("Unsupported file format. Use .xls or .xlsx")

TOL = 2  # ₹2 tolerance

tax_columns = ["cgst", "sgst", "igst"]

def money_matches(b, g):
    taxable_match = abs(b["taxable_value"] - g["taxable_value"]) <= TOL
    component_tax_match = all(
        abs(b[col] - g[col]) <= TOL
        for col in tax_columns
    )

    # Allows CGST+SGST in one file vs IGST in the other when total tax agrees.
    total_tax_match = abs(
        (b["cgst"] + b["sgst"] + b["igst"]) -
        (g["cgst"] + g["sgst"] + g["igst"])
    ) <= TOL

    return taxable_match and (component_tax_match or total_tax_match)

TOL = 2

books_r = books_grouped.copy()
gstr2b_r = gstr2b.copy()

# Branch 1: GSTIN + invoice must both match (design.md sheet 1A, first condition).
# This prevents a books row from matching a 2B row that merely shares its GSTIN.
merge_gstin_invoice = books_r.merge(
    gstr2b_r,
    left_on=["gstin", "Voucher Number"],
    right_on=["gstin", "invoice_number"],
    how="inner",
    suffixes=("_b", "_g")
)

# Branch 2: invoice + supplier must both match (design.md sheet 1A, second condition).
# Covers no-GSTIN rows and mistyped-GSTIN rows where invoice and supplier agree.
# This prevents a books row from matching a 2B row that merely shares its invoice number.
merge_invoice_supplier = books_r.merge(
    gstr2b_r,
    left_on=["Voucher Number", "Supplier"],
    right_on=["invoice_number", "supplier"],
    how="inner",
    suffixes=("_b", "_g")
)

candidates = pd.concat([merge_gstin_invoice, merge_invoice_supplier], ignore_index=True).drop_duplicates(
    subset=["books_group_idx", "gstr2b_idx"]
)

# Apply money tolerance filter. Tax buckets may match component-wise or by total tax.
taxable_match = abs(candidates["taxable_value_b"] - candidates["taxable_value_g"]) <= TOL
component_tax_match = (
    (abs(candidates["cgst_b"] - candidates["cgst_g"]) <= TOL) &
    (abs(candidates["sgst_b"] - candidates["sgst_g"]) <= TOL) &
    (abs(candidates["igst_b"] - candidates["igst_g"]) <= TOL)
)
total_tax_match = abs(
    (candidates["cgst_b"] + candidates["sgst_b"] + candidates["igst_b"]) -
    (candidates["cgst_g"] + candidates["sgst_g"] + candidates["igst_g"])
) <= TOL

sheet1_candidates = candidates[
    taxable_match & (component_tax_match | total_tax_match)
].copy()

# Ranking follows design.md section 7.
# invoice_exact: always True here (both branches require invoice to match), used as sanity check.
# gstin_exact: True only for branch-1 rows (joined on gstin); NaN for branch-2 rows after concat.
# supplier_exact: True for branch-2 rows (joined on supplier); may vary for branch-1 rows.
sheet1_candidates["invoice_exact"] = (
    sheet1_candidates["Voucher Number"] == sheet1_candidates["invoice_number"]
)
sheet1_candidates["gstin_exact"] = sheet1_candidates["gstin"].notna()
sheet1_candidates["supplier_exact"] = (
    sheet1_candidates["Supplier"] == sheet1_candidates["supplier"]
)
sheet1_candidates["total_money_diff"] = (
    (sheet1_candidates["taxable_value_b"] - sheet1_candidates["taxable_value_g"]).abs() +
    (sheet1_candidates["cgst_b"] - sheet1_candidates["cgst_g"]).abs() +
    (sheet1_candidates["sgst_b"] - sheet1_candidates["sgst_g"]).abs() +
    (sheet1_candidates["igst_b"] - sheet1_candidates["igst_g"]).abs()
)
sheet1_candidates["date_diff_days"] = (
    sheet1_candidates["date_b"] - sheet1_candidates["date_g"]
).abs().dt.days.fillna(999999)

complete_cols = ["gstin", "Voucher Number", "Supplier", "invoice_number", "supplier"]
sheet1_candidates["blank_count"] = sheet1_candidates[complete_cols].isna().sum(axis=1)

sheet1_candidates = sheet1_candidates.sort_values(
    by=[
        "invoice_exact",
        "gstin_exact",
        "supplier_exact",
        "total_money_diff",
        "date_diff_days",
        "blank_count",
    ],
    ascending=[False, False, False, True, True, True]
)

accepted_rows = []
needs_review_sheet1 = []
used_books = set()
used_gstr2b = set()

rank_cols = [
    "invoice_exact",
    "gstin_exact",
    "supplier_exact",
    "total_money_diff",
    "date_diff_days",
    "blank_count",
]

for books_id, group in sheet1_candidates.groupby("books_group_idx", sort=False):
    available = group[
        ~group["books_group_idx"].isin(used_books) &
        ~group["gstr2b_idx"].isin(used_gstr2b)
    ]

    if available.empty:
        continue

    best = available.iloc[0]
    tied_best = available[
        (available[rank_cols] == best[rank_cols]).all(axis=1)
    ]

    if len(tied_best) > 1:
        needs_review_sheet1.append(tied_best)
        continue

    accepted_rows.append(best)
    used_books.add(best["books_group_idx"])
    used_gstr2b.add(best["gstr2b_idx"])

sheet1a = pd.DataFrame(accepted_rows)
needs_review_sheet1 = (
    pd.concat(needs_review_sheet1, ignore_index=True)
    if needs_review_sheet1 else
    pd.DataFrame(columns=sheet1_candidates.columns)
)

books_pool = books_grouped[~books_grouped["books_group_idx"].isin(used_books)].copy()
gstr2b_pool = gstr2b[~gstr2b["gstr2b_idx"].isin(used_gstr2b)].copy()

print(f"Sheet 1A candidates: {len(sheet1_candidates)}")
print(f"Sheet 1A one-to-one matches: {len(sheet1a)}")
print(f"Sheet 1A needs review ties: {len(needs_review_sheet1)}")
print(f"Books groups remaining: {len(books_pool)}")

import re

# Reuse if Sheet 2 already ran, otherwise define here
try:
    standardize_invoice
except NameError:
    def standardize_invoice(val):
        if pd.isna(val) or str(val).strip() in ("", "<NA>", "nan"):
            return None
        s = re.sub(r"[\s/\-]+", "", str(val)).upper().lstrip("0")
        return s if s else None

# -------- SHEET 1B / 1C --------
# money + GSTIN  OR  money + supplier,  invoice does NOT match (exact or standardized)
# 1B: amount unique for that identity on both sides  →  auto-match
# 1C: amount not unique (collision)                  →  blue

books_1bc  = books_pool.copy()
gstr2b_1bc = gstr2b_pool.copy()
books_1bc["std_voucher"]  = books_1bc["Voucher Number"].apply(standardize_invoice)
gstr2b_1bc["std_invoice"] = gstr2b_1bc["invoice_number"].apply(standardize_invoice)

# Branch A: GSTIN identity
branch_a = books_1bc.merge(gstr2b_1bc, on="gstin", how="inner", suffixes=("_b", "_g"))
branch_a["match_branch"] = "gstin"

# Branch B: supplier identity
branch_b = books_1bc.merge(
    gstr2b_1bc,
    left_on="Supplier", right_on="supplier",
    how="inner", suffixes=("_b", "_g")
)
branch_b["match_branch"] = "supplier"
branch_b["gstin"] = branch_b["gstin_b"]   # normalize: "gstin" column always exists after concat

cands_1bc = pd.concat([branch_a, branch_b], ignore_index=True).drop_duplicates(
    subset=["books_group_idx", "gstr2b_idx"]
)

# Money filter
tv_ok   = abs(cands_1bc["taxable_value_b"] - cands_1bc["taxable_value_g"]) <= TOL
comp_ok = (
    (abs(cands_1bc["cgst_b"] - cands_1bc["cgst_g"]) <= TOL) &
    (abs(cands_1bc["sgst_b"] - cands_1bc["sgst_g"]) <= TOL) &
    (abs(cands_1bc["igst_b"] - cands_1bc["igst_g"]) <= TOL)
)
tot_ok  = abs(
    (cands_1bc["cgst_b"] + cands_1bc["sgst_b"] + cands_1bc["igst_b"]) -
    (cands_1bc["cgst_g"] + cands_1bc["sgst_g"] + cands_1bc["igst_g"])
) <= TOL
cands_1bc = cands_1bc[tv_ok & (comp_ok | tot_ok)].copy()

# Drop invoice-exact pairs — those belong in 1A / Sheet 2
inv_exact = cands_1bc["Voucher Number"] == cands_1bc["invoice_number"]
std_exact = (
    cands_1bc["std_voucher"].notna() &
    cands_1bc["std_invoice"].notna() &
    (cands_1bc["std_voucher"] == cands_1bc["std_invoice"])
)
cands_1bc = cands_1bc[~(inv_exact | std_exact)].copy()

# -------- Uniqueness: count (identity, taxable_value) occurrences in each pool --------
books_gstin_tv  = books_1bc.groupby(["gstin",    "taxable_value"])["books_group_idx"].count().to_dict()
books_supp_tv   = books_1bc.groupby(["Supplier", "taxable_value"])["books_group_idx"].count().to_dict()
gstr2b_gstin_tv = gstr2b_1bc.groupby(["gstin",   "taxable_value"])["gstr2b_idx"].count().to_dict()
gstr2b_supp_tv  = gstr2b_1bc.groupby(["supplier","taxable_value"])["gstr2b_idx"].count().to_dict()

def _get_counts(row):
    if row["match_branch"] == "gstin":
        b = books_gstin_tv.get( (row["gstin"],    row["taxable_value_b"]), 1)
        g = gstr2b_gstin_tv.get((row["gstin"],    row["taxable_value_g"]), 1)
    else:
        b = books_supp_tv.get(  (row["Supplier"], row["taxable_value_b"]), 1)
        g = gstr2b_supp_tv.get( (row["supplier"], row["taxable_value_g"]), 1)
    return pd.Series({"b_count": b, "g_count": g})

cands_1bc[["b_count", "g_count"]] = cands_1bc.apply(_get_counts, axis=1)
cands_1bc["is_unique"] = (cands_1bc["b_count"] == 1) & (cands_1bc["g_count"] == 1)

# -------- 1B: unique candidates → greedy one-to-one auto-match --------
unique_1bc = cands_1bc[cands_1bc["is_unique"]].copy()
unique_1bc["total_money_diff"] = (
    (unique_1bc["taxable_value_b"] - unique_1bc["taxable_value_g"]).abs() +
    (unique_1bc["cgst_b"]          - unique_1bc["cgst_g"]).abs()          +
    (unique_1bc["sgst_b"]          - unique_1bc["sgst_g"]).abs()          +
    (unique_1bc["igst_b"]          - unique_1bc["igst_g"]).abs()
)
unique_1bc["date_diff_days"] = (
    unique_1bc["date_b"] - unique_1bc["date_g"]
).abs().dt.days.fillna(999999)

unique_1bc = unique_1bc.sort_values(
    by=["total_money_diff", "date_diff_days"], ascending=[True, True]
)

accepted_1b    = []
used_books_1b  = set()
used_gstr2b_1b = set()
rank_cols_1b   = ["total_money_diff", "date_diff_days"]

for bid, group in unique_1bc.groupby("books_group_idx", sort=False):
    avail = group[
        ~group["books_group_idx"].isin(used_books_1b) &
        ~group["gstr2b_idx"].isin(used_gstr2b_1b)
    ]
    if avail.empty:
        continue
    best = avail.iloc[0]
    tied = avail[(avail[rank_cols_1b] == best[rank_cols_1b]).all(axis=1)]
    if len(tied) > 1:
        continue  # genuine tie → leave in pool, falls through to Sheet 2/3
    accepted_1b.append(best)
    used_books_1b.add(best["books_group_idx"])
    used_gstr2b_1b.add(best["gstr2b_idx"])

sheet1b = pd.DataFrame(accepted_1b)
if not sheet1b.empty:
    sheet1b["flag"] = "invoice_differs"
    sheet1b["step"] = "sheet1b"

# -------- 1C: collision candidates → blue --------
sheet1c_blue = cands_1bc[~cands_1bc["is_unique"]].copy()
if not sheet1c_blue.empty:
    sheet1c_blue["flag"] = "blue"
    sheet1c_blue["step"] = "sheet1c"

# Update global pools (only 1B matches are consumed; 1C rows stay for Sheet 2/3)
used_1b_books  = set(sheet1b["books_group_idx"]) if not sheet1b.empty else set()

    "Your job is to decide whether they are likely the SAME real-world invoice/supplier pair.\n\n"

    "IMPORTANT MATCHING PRINCIPLE:\n"
    "- Money is a hard gate: if taxable_diff or all tax differences are clearly outside tolerance, return no-match.\n"
    "- Money matching alone is NOT enough to match.\n"
    "- There must also be identity support from supplier name, invoice number, GST/business identity clues, or a strong abbreviation/legal-name relationship.\n"
    "- When uncertain, return no-match. A missed match is safer than a wrong match.\n\n"

    "FINANCIAL RULES:\n"
    "- Differences of 0, 1, or 2 rupees are acceptable rounding differences.\n"
    "- Small differences in CGST/SGST/IGST may be acceptable if total tax appears effectively the same.\n"
    "- If money differences are large or inconsistent, return likely_match=false even if names look similar.\n"
    "- Never use matching money by itself as the reason for a match.\n\n"

    "INVOICE NUMBER RULES:\n"
    "- Exact invoice number match is strong evidence.\n"
    "- Standardized invoice match is strong evidence: ignore spaces, hyphens, slashes, case, and leading zeros.\n"
    "- Prefix/suffix differences can still match when the rest is clearly the same, e.g. SPL-892 vs MSPL-892, FY/24-25/946 vs 946.\n"
    "- Internal voucher numbers may look completely different from the supplier invoice number; in that case supplier identity and money must be very strong.\n"
    "- If invoice numbers are different and supplier identity is weak, return no-match or low confidence.\n\n"

    "SUPPLIER NAME RULES — WHEN TO MATCH:\n"
    "- Known abbreviations or acronyms: L&T = Larsen and Toubro, HUL = Hindustan Unilever.\n"
    "- Only formatting differs: M/S prefix, Limited/Ltd suffix, & vs and, hyphen vs space, extra spaces.\n"
    "- Minor singular/plural variation: Hardware vs Hardwares.\n"
    "- Clear typo where the rest of the name is identical.\n"
    "- Transliteration variation of the same name: Sri vs Shri.\n"
    "- Bank entries: books may record a bank as 'KBL OD 0777000100021201' while GSTR-2B has 'KARNATAKA BANK LIMITED'. "
    "Known bank abbreviations include SBI, KBL, HDFC, ICICI, AXIS, BOI, PNB, BOB, CANARA, KOTAK, YES, IDFC, IOB, UCO, UBI.\n\n"

    "SUPPLIER NAME RULES — WHEN NOT TO MATCH:\n"
    "- Business-type word differs: TYRES vs MOTORS, ELECTRICALS vs PLYWOOD, STEEL vs CEMENT.\n"
    "- Short generic initials alone are not enough: NK, SR, RK with different suffixes.\n"
    "- Personal names with one-letter difference and no other support: SONA vs SOMA.\n"
    "- Common first word followed by different entity words: Akshaya Traders vs Akshaya Cement.\n\n"

    "CONFIDENCE GUIDANCE:\n"
    "- 0.95-1.00: supplier identity is clearly same and invoice/money also support it.\n"
    "- 0.85-0.94: strong likely match, minor invoice/name formatting issue.\n"
    "- 0.60-0.84: possible match but needs human review.\n"
    "- below 0.60: weak or unsafe match.\n\n"

    "Respond ONLY with valid JSON, no extra text, no markdown:\n"
    '{"likely_match": true or false, "confidence": 0.0 to 1.0, "reason": "one short sentence"}'

    "\n\nEXAMPLE OUTPUTS:\n"
    'For a confident match:\n'
    '{"likely_match": true, "confidence": 0.92, "reason": "Supplier abbreviation matches the legal name and invoice number is a close variant."}\n'
    'For a possible but unsafe match:\n'
    '{"likely_match": true, "confidence": 0.72, "reason": "Money matches and supplier names look related, but invoice numbers differ significantly."}\n'
    'For a no-match:\n'
    '{"likely_match": false, "confidence": 0.18, "reason": "Supplier business type and invoice number do not support the same entity."}\n\n'
    "Now respond ONLY with one valid JSON object for the given input."
)

# STEP 3: AI call function
# -----------------------------------------------------------------------

def clean_supplier_name(name):
    # Remove Excel carriage-return artifacts (_x000D_, _X000D_)
    name = re.sub(r'_[xX]000[dD]_', '', str(name))
    # Remove actual newlines and carriage returns
    name = name.replace('\r', '').replace('\n', '').strip()
    return name

async def ask_ai_sheet5(sem, row):
    async with sem:
        payload = {
            "books_supplier":  clean_supplier_name(str(row["Supplier"])),
            "2b_supplier":     clean_supplier_name(str(row["supplier"])),
            "books_invoice":   str(row["Voucher Number"]),
            "2b_invoice":      str(row["invoice_number"]),
            "taxable_diff":    round(float(row["taxable_value_b"] - row["taxable_value_g"]), 2),
            "cgst_diff":       round(float(row["cgst_b"]          - row["cgst_g"]),          2),
            "sgst_diff":       round(float(row["sgst_b"]          - row["sgst_g"]),          2),
            "igst_diff":       round(float(row["igst_b"]          - row["igst_g"]),          2),
        }
        user_content = json.dumps(payload, ensure_ascii=False)

        try:
            response = await client.chat.completions.create(
                model=AI_MODEL,
                messages=[
                    {"role": "system", "content": SYSTEM_PROMPT},
                    {"role": "user",   "content": user_content},
                ],
                temperature=0,
            )

            raw = response.choices[0].message.content.strip()
            clean = re.sub(r"^```[a-z]*\n?", "", raw).rstrip("` \n")
            clean = re.sub(r'[\x00-\x1f\x7f]', ' ', clean)

            try:
                parsed = json.loads(clean)
            except json.JSONDecodeError:
                m = re.search(r'\{.*?\}', clean, re.DOTALL)
                if m:
                    parsed = json.loads(m.group())
                else:
                    raise

            result = {
                "likely_match": bool(parsed.get("likely_match", False)),
                "confidence":   float(parsed.get("confidence", 0.0)),
                "reason":       str(parsed.get("reason", "")),
                "ai_input":     user_content,
                "raw_response": raw,
                "error":        None,
            }

        except Exception as e:
            result = {
                "likely_match": False,
                "confidence":   0.0,
                "reason":       "",
                "ai_input":     user_content,
                "raw_response": "",
                "error":        str(e),
            }

        return {**row.to_dict(), **result}



# -----------------------------------------------------------------------
# STEP 4: Run AI on every candidate pair
# -----------------------------------------------------------------------

sem = asyncio.Semaphore(10)
tasks = [ask_ai_sheet5(sem, row) for _, row in cands_5.iterrows()]

print(f"Sending {len(tasks)} pairs to AI (10 concurrent)...")
ai_results = await asyncio.gather(*tasks)
print("Done.")



# -----------------------------------------------------------------------
# STEP 5: Split results into proposals / blue / no-match
# -----------------------------------------------------------------------
# Sheet 5 does NOT consume rows from the pool here.
# It only produces proposals. Sheet 6 does the final greedy one-to-one merge.

sheet5_all = pd.DataFrame(ai_results)
sheet5_all["step"] = "sheet5_ai"

# High-confidence positive → proposal for Sheet 6 to accept
sheet5_proposals = sheet5_all[
    sheet5_all["likely_match"] & (sheet5_all["confidence"] >= AI_CONF_THRESHOLD)
].copy()
sheet5_proposals["flag"] = "ai_match"

# Low-confidence positive → human should confirm before we seal it
sheet5_blue = sheet5_all[
    sheet5_all["likely_match"] & (sheet5_all["confidence"] < AI_CONF_THRESHOLD)
].copy()
sheet5_blue["flag"] = "blue"

# AI said no → stays in unmatched pool, Sheet 6 will list it
sheet5_no_match = sheet5_all[~sheet5_all["likely_match"]].copy()

# Any API errors — print so you can diagnose (wrong key, wrong model name, etc.)
sheet5_errors = sheet5_all[sheet5_all["error"].notna()].copy()

print(f"\nSheet 5 results:")
print(f"  Proposals (confident AI match): {len(sheet5_proposals)}")
print(f"  Blue (low confidence):          {len(sheet5_blue)}")
print(f"  No match:                       {len(sheet5_no_match)}")
if len(sheet5_errors) > 0:
    print(f"\n  ⚠ Errors ({len(sheet5_errors)} calls failed) — check API key / model ID:")
    print(sheet5_errors[["Supplier", "supplier", "error"]].to_string())
